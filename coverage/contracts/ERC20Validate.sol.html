pragma solidity ^0.6.0;
 
library ERC20Validate {
    function beforeApproval(address _owner, address _spender)
        external
        pure
        returns (uint256)
    {
        require(
            _owner != address(0) && _spender != address(0),
            "ERC20: APPROVE_BAD_CONDITIONS"
        );
    }
 
    function beforeMint(address account) external pure returns (uint256) {
        require(account != address(0), "ERC20: mint to the zero address");
    }
 
    function beforeBurn(address account) external pure returns (uint256) {
        require(account != address(0), "ERC20: burn from the zero address");
    }
 
    function beforeTransfer(address sender, address recipient)
        external
        pure
        returns (uint256)
    {
        require(
            sender != address(0) && recipient != address(0),
            "ERC20: TRANSFER_BAD_CONDITIONS"
        );
    }
}
