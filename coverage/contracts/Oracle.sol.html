/**
 * @title Oracle
 * @author @kohshiba
 * @notice This contract is the functory contract that manages functions related to market creation activities.
 */
 
pragma solidity ^0.6.0;
pragma experimental ABIEncoderV2;
 
import "./libraries/math/SafeMath.sol";
import "./libraries//utils/Ownable.sol";
import "./libraries/utils/Address.sol";
import "./interfaces/IPoolTemplate.sol";
 
contract Oracle is Ownable {
    using SafeMath for uint256;
    using Address for address;
    struct Incident {
        uint256 payoutNumerator;
        uint256 payoutDenominator;
        uint256 incidentTimestamp;
    }
    mapping(address => Incident) public incidents;
 
    constructor() public {}
 
    function applyCover(
        address _target,
        uint256 _payoutNumerator,
        uint256 _payoutDenominator,
        uint256 _incidentTimestamp
    ) external {
        require(address(_target) != address(0));
        IPoolTemplate(_target).applyCover(now.add(10 days));
        incidents[_target].payoutNumerator = _payoutNumerator;
        incidents[_target].payoutDenominator = _payoutDenominator;
        incidents[_target].incidentTimestamp = _incidentTimestamp;
    }
 
    function reportIncident(address _target, uint256 _incidentTimestamp)
        external
    {
        IPoolTemplate(_target).reportIncident(
            now.add(3 days),
            _incidentTimestamp
        );
    }
 
    function getClaimDetails()
        external
        view
        returns (
            uint256,
            uint256,
            uint256
        )
    {
        return (
            incidents[msg.sender].payoutNumerator,
            incidents[msg.sender].payoutDenominator,
            incidents[msg.sender].incidentTimestamp
        );
    }
}
